let userName = ''; let randomNumber = Math.floor(Math.random() * 100) + 1; function submitName() { userName = document.getElementById('name-input').value; if (userName) { document.getElementById('user-name').innerText = `مرحبًا، ${userName}!`; document.getElementById('user-name').style.display = 'block'; document.getElementById('game-area').style.display = 'block'; } } function checkGuess() { const guess = parseInt(document.getElementById('guess').value); const messageElement = document.getElementById('message'); if (guess === randomNumber) { messageElement.innerText = 'تهانينا! لقد خمنت الرقم الصحيح!'; messageElement.style.color = '#4CAF50'; } else if (guess < randomNumber) { messageElement.innerText = 'الرقم أكبر من تخمينك. حاول مرة أخرى!'; } else { messageElement.innerText = 'الرقم أصغر من تخمينك. حاول مرة أخرى!'; } }
